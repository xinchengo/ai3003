\section{问题一}

\begin{align*}
  & \quad \frac{\partial \mathcal L_{\text{rec}}}{\partial w_\mu} = \frac{\partial \mathcal L_{\text{rec}}}{\partial \hat x} \cdot \frac{\partial \hat{x}}{\partial z} \cdot \frac{\partial z}{\partial \mu} \cdot \frac{\partial \mu}{\partial w_\mu} \\
  & \text{其中} z \sim \mathcal N(\mu, \sigma^2) \\
  & \implies \frac{\partial z}{\partial \mu}~\text{是没有定义的，无法计算梯度}
\end{align*}

\section{问题二}

\begin{align*}
  & \text{此时的关键在于 {\it 确定了 z 的生成方式}} \\
  & \quad z = \mu + \sigma \cdot \epsilon \implies \frac{\partial z}{\partial \mu} = 1 \\
  &
  \begin{aligned}
    \text{而}&~\mathcal L_{\text{rec}} = \frac{1}{2}(x - \hat x)^2 \implies \frac{\partial \mathcal L_{\text{rec}}}{\partial \hat x} = \hat x - x \\
    &~\hat x = az + c \implies \frac{\partial \hat x}{\partial z} = a \\
    &~\mu = w_\mu x + b_\mu \implies \frac{\partial \mu}{\partial w_\mu} = x
  \end{aligned} \\
  & \implies \frac{\partial \mathcal L_{\text{rec}}}{\partial w_\mu} = (\hat x - x) \cdot a \cdot 1 \cdot x = a(\hat x - x)x
\end{align*}
